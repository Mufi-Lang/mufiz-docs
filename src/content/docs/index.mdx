import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
	<Card title="Built Under Zig" icon="seti:zig">
	Mufi-Lang's compiler, MufiZ, is built under Zig programming language.
	This allows us to be able to easily cross-compile MufiZ to different toolchains,
	including Windows, Linux, and MacOS. The language is built to be memory-safe, fast
	and robust, with excellent C-interoperability (which we still depend upon).
	</Card>
	<Card title="Experimental" icon="warning">
	Mufi-Lang is an experimental language and has not reached a stable `v1.0` release.
	As a solo developer working on this project, I am trying my best to grow the language,
	however it is still in its early stages, and we hope to incorporate more research
	towards newer features and improvements.
	</Card>
	<Card title="Opensource Forever" icon="openCollective">
	Mufi-Lang's compiler, MufiZ is opensource and licensed under the GPLv2 license,
	and a project under the MoKa Reads Foundation, a collective under the OpenCollective.
	</Card>
